<section id="client-settings">
  <h2><i class="fas fa-cog"></i> MegaMek Client Settings</h2>

  <p>Access via <strong>View &gt; Client Settings</strong> in MegaMek.</p>

  <h3>Recommended Settings</h3>

  <table class="guide-table">
    <thead>
      <tr>
        <th>Setting</th>
        <th>Recommendation</th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td><strong>Skip 'Done' when firing all weapons</strong></td>
        <td>Turn off to review targets carefully</td>
      </tr>
      <tr>
        <td><strong>Show damage to units on unit label</strong></td>
        <td>Enable for quick damage assessment</td>
      </tr>
      <tr>
        <td><strong>Default Weapon Sort Order</strong></td>
        <td>Highest to lowest damage</td>
      </tr>
      <tr>
        <td><strong>Minimap</strong></td>
        <td>Choose your preferred theme</td>
      </tr>
      <tr>
        <td><strong>Auto Display</strong></td>
        <td>Keep Unit Display docked, hide Force Display</td>
      </tr>
    </tbody>
  </table>

  <h3>GUI Scaling</h3>

  <p>The entire interface can be scaled to fit your display. Tooltip fonts can be adjusted separately in the <strong>Unit Tooltip</strong> tab.</p>
</section>

<section id="shortcuts">
  <h2><i class="fas fa-keyboard"></i> Keyboard Shortcuts</h2>

  <h3>View Controls</h3>

  <table class="guide-table">
    <thead>
      <tr>
        <th>Shortcut</th>
        <th>Action</th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td><kbd>T</kbd></td>
        <td>Toggle isometric view (helpful for elevation)</td>
      </tr>
      <tr>
        <td><kbd>C</kbd></td>
        <td>Toggle visual/sensor range display</td>
      </tr>
      <tr>
        <td><kbd>Ctrl</kbd>+<kbd>R</kbd></td>
        <td>Show round report</td>
      </tr>
      <tr>
        <td><kbd>Ctrl</kbd>+<kbd>P</kbd></td>
        <td>Show/hide weather and planetary conditions</td>
      </tr>
      <tr>
        <td><kbd>Middle Click</kbd> (hex 1, then hex 2)</td>
        <td>Measure distance and check LOS</td>
      </tr>
    </tbody>
  </table>

  <h3>Movement Controls</h3>

  <table class="guide-table">

  <table class="guide-table">

<section id="chat">
  <h2><i class="fas fa-comments"></i> Chat Commands</h2>

  <p>Press <kbd>Enter</kbd> to open the chat box, then type:</p>

  <table class="guide-table">
    <thead>
      <tr>
        <th>Command</th>
        <th>Effect</th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td><code>/victory</code></td>
        <td>Makes OpFor bots surrender (fast scenario resolution)</td>
      </tr>
      <tr>
        <td><code>/defeat</code></td>
        <td>Concede the battle</td>
      </tr>
      <tr>
        <td><code>/gm</code></td>
        <td>Grant GM rights for the current game</td>
      </tr>
      <tr>
        <td><code>/help</code></td>
        <td>List all available commands</td>
      </tr>
    </tbody>
  </table>
</section>

<section id="loop">
  <h2><i class="fas fa-sync-alt"></i> Gameplay Loop Summary</h2>

  <p>The core MekHQ campaign loop:</p>

  <div class="row">
    <div class="col-lg-6">
      <div class="guide-step">
        <div class="step-number">1</div>
        <div class="step-content">
          <p><strong>Find a contract</strong> in the Contract Market</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">2</div>
        <div class="step-content">
          <p><strong>Travel</strong> to the contract location</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">3</div>
        <div class="step-content">
          <p><strong>Deploy forces</strong> according to contract requirements</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">4</div>
        <div class="step-content">
          <p><strong>Scout</strong> the area with Patrol forces</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">5</div>
        <div class="step-content">
          <p><strong>Complete scenarios</strong> as they appear</p>
        </div>
      </div>
    </div>
    <div class="col-lg-6">
      <div class="guide-step">
        <div class="step-number">6</div>
        <div class="step-content">
          <p><strong>Repair and resupply</strong> between battles (manage {% include glossary_term.html id="fatigue" text="fatigue" %})</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">7</div>
        <div class="step-content">
          <p><strong>Achieve objectives</strong> to earn {% include glossary_term.html id="contract_victory_points" text="Victory Points" %}</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">8</div>
        <div class="step-content">
          <p><strong>Break enemy morale</strong> to win the contract</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">9</div>
        <div class="step-content">
          <p><strong>Resolve the contract</strong> and collect payment</p>
        </div>
      </div>

      <div class="guide-step">
        <div class="step-number">10</div>
        <div class="step-content">
          <p><strong>Return to base</strong> or accept a new contract</p>
        </div>
      </div>
    </div>
  </div>

  <h2><i class="fas fa-feather-alt"></i> Final Notes</h2>

  <div class="guide-callout guide-callout-info">
    <div class="callout-title"><i class="fas fa-book-open"></i> Storytelling Tool</div>
